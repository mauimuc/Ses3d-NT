116  ! Passed dummy arguments

118  REAL(real_kind), INTENT(IN) :: min, max
119  ! Result
120  REAL(real_kind) :: coords(elms, lpd+1)
121  ! Local variables
122  REAL(real_kind) :: knots(lpd+1), delta
123  INTEGER :: i
124 
125  ! Retrieve GLL Points
126  knots(:) = get_knots(lpd)
127  ! Scale GLL Points to be in the interval [0,1]
128  knots(:) = 0.5_real_kind*(1.0_real_kind + knots(:))
129 
130  ! Calculate element width

132 
133  ! Make coordinate vector
134  DO i=0, elms - 1
135  coords(i+1,:) = min + (i + knots(:))*delta
