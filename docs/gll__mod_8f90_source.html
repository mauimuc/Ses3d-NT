251  ELEMENTAL FUNCTION get_dlgll(n,i,j) RESULT(y)
252  INTEGER, INTENT(IN) :: n, i, j
253  REAL(rk) :: y
254  REAL(rk), ALLOCATABLE :: knots(:), x_k(:)
255  REAL(rk) :: x_i, x_j
256 
257  ! Determine collocation points (knots)
258  ALLOCATE( knots(0:n) )
259  knots(:) = get_knots( n )
260 
261  x_i = knots(i)
262  x_j = knots(j)
263 
264  ! Pop x_i
265  x_k = pack( knots, knots/=x_i )
266 
267  ! Compute derivative of LAGRANGE polynomial i at collocation point j
268  IF ( i==j ) THEN
269  y = sum( 1.0 / ( x_i-x_k(:) ) )
270  ELSE
271  y = product( x_j-x_k(:), mask=( x_k(:)/=x_j ) ) / &
272  product( x_i-x_k(:) )
273  ENDIF

275  END FUNCTION get_dlgll
