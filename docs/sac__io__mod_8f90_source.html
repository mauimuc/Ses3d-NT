229  FUNCTION read_sac(file_name) RESULT(self)
230  use, INTRINSIC :: iso_fortran_env, only : real32, int32
231  TYPE(sac_cls) :: self
232  CHARACTER(*), INTENT(IN) :: file_name
233  INTEGER :: lun
234  REAL(real32) :: floats_(1:70)
235  INTEGER(INT32) :: ints_(71:105), npts_, iftype_
236  LOGICAL(4) :: bools_(106:110), leven_
237  CHARACTER(LEN=4,KIND=ascii) :: chars_(111:158)
238  REAL(real32), ALLOCATABLE :: data1_(:), data2_(:)
239 
240  ! Check boolean storage size
241  IF ( storage_size(bools_) /= 32 ) &
242  stop 'Logical sotrage size /= 32'
243 
244  OPEN(newunit=lun, file=file_name, access='stream', action='read', &
245  form='unformatted', status='old' )
246 
247  ! Read file header, the first 158 words
248  ! REAL32, INTEGER32, LOGICAL XXX, ASCII
249  READ(unit=lun) floats_
250  READ(unit=lun) ints_
251  READ(unit=lun) bools_
252  READ(unit=lun) chars_
253 
254  ! Number of sampling points
255  npts_ = ints_( header('NPTS') )
256  ! See if data are evenly spaced
257  leven_ = bools_( header('LEVEN') )
258  ! See which type it is e.g. spectral data
259  iftype_ = ints_( header('IFTYPE') )
260 
261  ! Allocate memory to for the dependent variable
262  ALLOCATE( data1_(npts_) )
263  ! Read First Data Section
264  ! REAL32
265  READ( unit=lun ) data1_
266 
267  ! Handle independent variable
268  IF ( .NOT. leven_ &
269  .OR. iftype_==enum('IRLIM') & ! IRLIM {Spectral file---real and imaginary}
270  .OR. iftype_==enum('IAMPH') ) THEN ! IAMPH {Spectral file---amplitude and phase}
271  ALLOCATE( data2_(npts_) )
272  ! Read Second Data Section
273  ! REAL32
274  READ( unit=lun ) data2_
275  !self%data2(:) = data2_
276  ELSE
277  ! Not needed; just allocate an empty array
278  ALLOCATE( data2_(0) )
279  END IF
280 
281  ! All don; close unit identifier
282  CLOSE( unit=lun )
283 
284  ! Invoke the default structure constructor
285  self = sac_cls(floats=floats_, ints=ints_, bools=bools_, &
286  chars=chars_, data1=data1_, data2=data2_)
287 
288  END FUNCTION read_sac

453  SUBROUTINE write_sac(self, file, iostat, iomsg)
454  use, INTRINSIC :: iso_fortran_env, only : real32, int32
455  CLASS(sac_cls), INTENT(IN) :: self
456  CHARACTER(*), INTENT(IN) :: file
457  CHARACTER(*), INTENT(OUT), OPTIONAL :: iomsg
458  INTEGER, INTENT(OUT), OPTIONAL :: iostat
459  CHARACTER(7) :: status
460  INTEGER :: lun
461 
462  ! Determine file status at open
463  IF ( self%bools(header('LOVROK')) ) THEN
464  status = 'replace' ! Override existing files
465  ELSE
466  status = 'new' ! DO NOT override
467  END IF
468 
469  ! Open file depending on optional arguments iostat and iomsg
470  IF ( PRESENT(iostat) .AND. PRESENT(iomsg) ) THEN
471  OPEN( newunit=lun, file=file, access='stream', form='unformatted', &
472  action='write', status=status, iostat=iostat, iomsg=iomsg )
473  IF ( iostat /= 0 ) RETURN
474  ELSE IF( PRESENT(iostat) ) THEN
475  OPEN( newunit=lun, file=file, access='stream', form='unformatted', &
476  action='write', status=status, iostat=iostat )
477  IF ( iostat /= 0 ) RETURN
478  ELSE
479  OPEN( newunit=lun, file=file, access='stream', form='unformatted', &
480  action='write', status=status )
481  END IF
482 
483  ! Write header
484  WRITE( unit=lun ) real( self%floats, kind=real32 )
485  WRITE( unit=lun ) int( self%ints , kind=int32 )
486  WRITE( unit=lun ) LOGICAL( self%bools , KIND=logical_kind )
487  WRITE( unit=lun ) self%chars
488  ! Write data
489  IF ( ALLOCATED( self%data1 ) ) &
490  WRITE( unit=lun ) real( self%data1, kind=real32 )
491  IF ( ALLOCATED( self%data2 ) ) &
492  WRITE( unit=lun ) real( self%data2, kind=real32 )
493 
494  ! Close file
495  CLOSE( lun )
496 
497  IF( PRESENT(iostat) ) &
498  iostat = 0
499  IF( PRESENT(iomsg) ) &
500  iomsg = "File '" // trim(file) // "' successfully written"
501 
502  END SUBROUTINE write_sac
