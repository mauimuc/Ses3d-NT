158  ENDDO
159  ENDDO

161  !-------------------------------------------------------------------------
162  !- complete velocity gradients in spherical coordinates ------------------
163  !-------------------------------------------------------------------------
164

166  eyy(:,:,:,:,:,:) = eyy/(r_sin_theta)+(vz+vx*cot_theta)/r ! (grad v)_(phi phi)
167  exx(:,:,:,:,:,:) = (vz+exx)/r ! (grad v)_(theta theta)
168  ezx(:,:,:,:,:,:) = -ezx ! (grad v)_(r theta)
169  exz(:,:,:,:,:,:) = (exz-vx)/r ! (grad v)_(theta r)
170  ezy(:,:,:,:,:,:) = -ezy ! (grad v)_(r phi)

172  exy(:,:,:,:,:,:) = exy/r ! (grad v)_(theta phi)
173  eyx(:,:,:,:,:,:) = -vy*cot_theta/r+eyx/(r_sin_theta) ! (grad v)_(phi theta)
174 
175  !--------------------------------------------------------------------------
176  !- integrate velocity gradients to displacement gradients -----------------
177  !--------------------------------------------------------------------------
178

180  dyuy(:,:,:,:,:,:)=dyuy+dt*eyy
181  dxux(:,:,:,:,:,:)=dxux+dt*exx
182  dzux(:,:,:,:,:,:)=dzux+dt*ezx
183  dxuz(:,:,:,:,:,:)=dxuz+dt*exz
184  dzuy(:,:,:,:,:,:)=dzuy+dt*ezy
185  dyuz(:,:,:,:,:,:)=dyuz+dt*eyz

427
