151  PURE FUNCTION lagrange_p_kw(x, knots, weights) RESULT(l)
152  REAL(rk), INTENT(IN) :: x, knots(:), weights(:)
153  REAL(rk) :: l(size(knots))
154  LOGICAL :: mask(size(knots))
155 
156  IF ( SIZE( knots ) .LE. 1 ) THEN ! only one sampling point
157  l(:) = 1.0
158  RETURN
159  END IF
160 
161  mask(:) = x .NE. knots
162  IF ( all( mask ) ) THEN ! if collocation-p. != interpolation-p.
163  l(:) = product( x - knots(:) ) / ( x - knots(:) ) * weights
164  RETURN
165  ELSE ! if interpolation-p. matches with one collocation-p.
166  ! uses the property l_j(x_i) = delta_ij
167  l(:) = merge( 0.0_rk, 1.0_rk, mask )
168  RETURN
