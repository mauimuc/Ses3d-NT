433  SUBROUTINE parse_receiver_file( receiver_file_name, receivers )
434  USE parser_mod, ONLY : count_namelists, parse_namelist, &
435  receiver_typ
436  USE bcast_mod, ONLY : bcast_namelist, bcast_integer, reduce_sum
437  ! Passed dummy arguments
438  CHARACTER(LEN=*), INTENT(IN) :: receiver_file_name
439  TYPE( receiver_cls ), INTENT(OUT), ALLOCATABLE :: receivers(:)
440  ! Local variables
441  INTEGER :: receiver_file_unit
442  TYPE( receiver_typ) :: receiver
443  INTEGER :: n_recs, i
444 
445  ! Open receiver file
446  IF ( is_mpi_master_rank() ) &
447  OPEN( newunit=receiver_file_unit, file=receiver_file_name, &
448  action='READ', status='OLD' )
449 
450  !----------------
451  ! Parse receivers
452  !----------------
453  IF ( is_mpi_master_rank() ) &
454  ! Count number of NAMELIST groups
455  n_recs = count_namelists( unit=receiver_file_unit, nml_cls=receiver)
456 
457  ! Broadcast n_recs amongst all ranks
458  CALL bcast_integer(i=n_recs, mr=mpi_master_rank)
459 
460  ! Allocate array of receivers
461  ALLOCATE( receivers(n_recs) )
462 
463  IF ( is_mpi_master_rank() ) &
464  rewind( unit=receiver_file_unit )
465 
466  ! Iterate through all &receiver groups
467  DO i=1, n_recs
468  IF ( is_mpi_master_rank() ) THEN
469  ! Parse NAMELIST
470  CALL parse_namelist(unit=receiver_file_unit, nml_cls=receiver)
471  END IF
472  ! Broadcast values
473  CALL bcast_namelist(receiver, mpi_master_rank)
474  ! Invoke structure constructor
475  receivers(i) = receiver_cls(receiver)
476  END DO
477 
478  ! all done -> close file
479  IF ( is_mpi_master_rank() ) &
480  CLOSE( receiver_file_unit )
481 
482  ! Sort out all receivers which are not located in this rank
483  receivers = pack( receivers(:), .NOT. receivers(:)%uninitialized() )
484 
485  ! Check if there is at least one receiver station left
486  n_recs = reduce_sum(i=SIZE(receivers), mr=mpi_master_rank)
487  IF ( n_recs < 1 .AND. is_mpi_master_rank() ) &
488  CALL warn( 'No receivers defined' )
