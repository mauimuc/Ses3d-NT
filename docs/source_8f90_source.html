564  FUNCTION make_source_signature_(wavelet, onset, width) &
565  result(signature)
566  ! Use associated entities
567  USE error_mod, ONLY : warn, abort
568  USE source_signature_mod, ONLY : delta, ricker, sin_3, gauss, brune
569  ! Passed dummy variables
570  CHARACTER(LEN=*), INTENT(IN) :: wavelet
571  REAL(real_kind), INTENT(IN) :: onset, width
572  ! Result variable
573  REAL(real_kind) :: signature(config%nt())
574  ! Local variables
575  REAL(real_kind) :: t(config%nt()), dt
576  INTEGER :: i, nt
577 
578  nt = config%nt()
579  dt = config%dt()
580 
581  ! Make time vector
582  t(:) = real([ (i, i=0, nt-1) ], real_kind)*dt
583 
584  ! Select the source signature
585  SELECT CASE ( trim(wavelet) )
586 
587  ! Excite with delta-pulse
588  CASE ( 'DELTA' )
589  ! Check if parameter onset is appropriate
590  IF ( .NOT. any(t(:)==onset) ) &
591  CALL warn("For a DELTA-pulse the value of 'onset' must pre&
592  &cisely coinside with a time-step (n*dt). This is not &
593  &the case! The entire exciting wavelet is set to zero.")
594  ! Make source signature
595  signature(:) = delta(time=t(:), onset=onset)
596 
597  ! Excite with Ricker-wavelet
598  CASE ( 'RICKER' )
599  ! Check if parameters onset and width are appropriate
600  IF ( t(1) > onset .OR. t(nt) < onset ) &
601  CALL warn("For a RICKER-wavelet the value of 'onset' should&
602  & be in the range of in the simulation time.")
603  IF ( width < 3*dt .OR. width > t(nt/2) ) &
604  CALL warn("The RICKER-wavelet's width appears to be either &
605  &too wide or too narrow.")
606  ! Make source signature
607  signature(:) = ricker( time=t(:), onset=onset, width=width)
608 
609  ! Excite with Gaussian pulse
610  CASE ( 'GAUSS' )
611  ! Check if parameters onset and width are appropriate
612  IF ( t(1) > onset .OR. t(nt) < onset ) &
613  CALL warn("For a GAUSS-puls the value of 'onset' should&
614  & be in the range of in the simulation time.")
615  IF ( width < 3*dt .OR. width > t(nt/2) ) &
616  CALL warn("The GAUSS-puls' width appears to be either &
617  &too wide or too narrow.")
618  ! Make source signature
619  signature(:) = gauss(time=t(:), onset=onset, width=width)
620 
621  ! Excite with sin()**3 wavelet
622  CASE ( 'SIN**3' )
623  ! Check if parameters onset and width are appropriate
624  IF ( t(1) > onset .OR. t(nt) < onset ) &
625  CALL warn("For a SIN**3-puls the value of 'onset' should&
626  & be in the range of in the simulation time.")
627  IF ( width < 3*dt .OR. width > t(nt/2) ) &
628  CALL warn("The SIN**3-puls' width appears to be either &
629  &too wide or too narrow.")
630  ! Make source signature
631  signature(:) = sin_3(time=t(:), onset=onset, width=width)
632 
633  ! Excite with Brune signature
